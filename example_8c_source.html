187 
188  /* The agent information to return. */
189  unsigned int ag_info = 0;
190 
191  /* Convert generic pointer to world to a WORLD object. */
192  WORLD *my_world = (WORLD *) w;
193 
194  /* Check if the given (x,y) coordinates are within bounds of the world. */
195  if ((x >= my_world->xsize) || (y >= my_world->ysize)) {
196 
197  /* If we got here, then the coordinates are off bounds. As such we will
198  report that the requested agent is of unknown type. No need to
199  specify agent ID or playable status, since the agent is unknown. */
200  ag_info = Unknown;
201 
202  } else {
203 
204  /* Given coordinates are within bounds, let's get and pack the request
205  agent information. */
206 
207  /* Obtain agent at specified coordinates. */
208  AGENT ag = my_world->grid[x * my_world->xsize + y];
209 
210  /* Is there an agent at (x,y)? */
211  if (ag.type == None) {
212 
213  /* If there is no agent at the (x,y) coordinates, set agent type to
214  None. No need to specify agent ID or playable status, since
215  there is no agent here. */
216  ag_info = None;
217 
218  } else {
219 
220  /* If we get here it's because there is an agent at (x,y). Bit-pack
221  all the agent information as specified by the get_agent_info_at
222  function pointer definition. */
223  ag_info = (ag.id << 3) | (ag.playable << 2) | ag.type;
224 
225  }
226 
227  }
228
